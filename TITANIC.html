# Estimar parmetros de la distribucin normal
mu, sigma = np.mean(numeric_df['Survived']), np.std(numeric_df['Survived'])

# Crear el rango de valores para la curva
x = np.linspace(min(numeric_df['Survived']), max(numeric_df['Survived']), 100)

y = stats.norm.pdf(x, mu, sigma)

# Graficar el histograma y la curva de densidad
plt.hist(numeric_df['Survived'], bins=30, density=True, alpha=0.6, color='b', label='Histograma de datos')
plt.plot(x, y, 'r', label=f'Distribucin Normal ($\mu={mu:.2f}$, $\sigma={sigma:.2f}$)')
plt.xlabel('Valores')
plt.ylabel('Densidad de Probabilidad')
plt.title('Aproximacin a la Distribucin Normal de Sobrevivientes')
plt.legend()
plt.show()
